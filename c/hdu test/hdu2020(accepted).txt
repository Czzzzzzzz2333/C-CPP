#include<stdio.h>
#include<math.h>

int main()
{
    int n,temp;
    while(scanf("%d",&n)!=EOF&&n<=100)
    {
        if(n==0)
            break;
        int a[n];
        for(int i=0; i<n; i++)
        {
            scanf("%d",&a[i]);
        }
        for(int i=0; i<n; i++)
        {
            for(int j=i; j<n; j++)
            {
                if(fabs(a[j])>=fabs(a[i]))
                {
                    temp=a[i];
                    a[i]=a[j];
                    a[j]=temp;
                }
            }
        }
        printf("%d",a[0]);//a[0]并没有交换
        for (int i = 1; i < n; i++)
        {
            printf(" %d",a[i]);
        }
        printf("\n");
    }
    return 0;
}
