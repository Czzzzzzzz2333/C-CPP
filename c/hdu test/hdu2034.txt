#include<stdio.h>

int main()
{
    int n,m;
    while(~scanf("%d%d",&n,&m))
    {
        if(n==0&&m==0)
            break;
        int s1[105],s2[105],l[105],count=0,temp=0,h=0;
        for(int i=1; i<=n; i++)
            scanf("%d",&s1[i]);
        for(int i=1; i<=m; i++)
            scanf("%d",&s2[i]);
        for(int i=1; i<=n; i++)
        {
            for(int j=1; j<=m; j++)
            {
                if(s1[i]!=s2[j])
                    count++;
            }
            if(count==m)
            {
                l[++h]=s1[i];
                temp=1;
            }
            count=0;
        }
        if(temp==0)
            printf("NULL\n");
        else
        {
            int i,j;
            for(i=1; i<h; i++)
                for(j=1; j<=h-i; j++)
                    if(l[j]>l[j+1])
                    {
                        l[j]^=l[j+1];
                        l[j+1]^=l[j];
                        l[j]^=l[j+1];
                    }
            for(i=1; i<=h-1; i++)
                printf("%d ",l[i]);
            printf("%d \n",l[i]);
        }
    }
    return 0;
}
