#include<stdio.h>
int mod_pow(int m, int n,int mod){      //¿ìËÙÃÝ
    int res = 1;
    while( n > 0 ){
        if(n%2==1) res = res * m % mod;
        m = m * m % mod;
        n/=2;
    }
    return res;
}
int main(){ 
    int m,n;
    while(scanf("%d%d",&m,&n),n||m)
        printf("%d\n",mod_pow(m,n,1000));
    return 0;
}
