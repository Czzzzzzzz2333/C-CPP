#include<stdio.h>

#define MAX 1000
int father[1000],count;

void Init(int n);
int Find(int s);
int Combine(int x,int y);
void Init(int n)
{
    for(int i=1; i<=n; i++)
        father[i]=i;
}

int Find(int s)
{
    while(father[s]!=s)
    {
        s=father[s];
    }
    return s;
}

int Combine(int x,int y)
{
    father[x]=y;
    return ++count;
}

int main()
{
    int n,m;
    while(~scanf("%d",&n))
    {
        if(!n)
            break;
        count=0;
        Init(n);
        scanf("%d",&m);
        for(int i=0; i<m; i++)
        {
            int a,b,x,y;
            scanf("%d%d",&a,&b);
            x=Find(a);
            y=Find(b);
            if(x!=y)
            {
                Combine(x,y);
            }
        }
        printf("%d\n",n-count-1);
    }
    return 0;
}
