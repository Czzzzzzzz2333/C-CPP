#include<stdio.h>

int fun(int m);

int  main()
{
    int n,m,sum;
    scanf("%d",&n);
    getchar();
    while(n--)
    {
        scanf("%d",&m);
        sum=fun(m);
        printf("%d\n",sum);
    }
    return 0;
}

int fun(int m)
{
    if(m<=2)
        return 1;
    else
        return fun(m-1)+fun(m-2);
}
